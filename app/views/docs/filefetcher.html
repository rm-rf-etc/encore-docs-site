FileFetcher

FileFetcher is a function that finds files matching your rules, and passes the full path to the callback that you provide. Available through npm. 

var fetchfiles = require('filefetcher')

fetchfiles([
    { path:'./app/controllers', recursive:true, type:'js',   cb:console.log },
    { path:'./app/models',      recursive:true, type:'js',   cb:console.log },
    { path:'./app/views',       recursive:true, type:'html', cb:console.log }
])
